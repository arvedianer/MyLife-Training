Replace current `calculateStreak(dates: string[])` signature with:
```ts
export function calculateStreak(workoutDates: string[], restDays: string[] = []): number {
  const allDates = Array.from(new Set([...workoutDates, ...restDays]));
  if (allDates.length === 0) return 0;
  const sorted = allDates.sort().reverse();
  let streak = 0;
  let cursor = new Date();
  cursor.setHours(0, 0, 0, 0);

  for (const dateStr of sorted) {
    const parts = dateStr.split('-');
    const date = new Date(Number(parts[0]), Number(parts[1]) - 1, Number(parts[2]));
    const diff = Math.round((cursor.getTime() - date.getTime()) / (1000 * 60 * 60 * 24));
    if (diff <= 1) {
      streak++;
      cursor = date;
    } else {
      break;
    }
  }
  return streak;
}
```
